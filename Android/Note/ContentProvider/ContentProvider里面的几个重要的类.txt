private static final int PEOPLE = 1;

sURIMatcher.addURI("contacts", "people/#/phones/#", PEOPLE_PHONES_ID);   //#代表数字
sURIMatcher.addURI("call_log", "calls/filter/*", CALLS_FILTER);			 //*代表任意字符
int match = sURIMatcher.match(url);  				  					 //开始匹配
